**Goal:** Add physically-based Image-Based Lighting (split-sum IBL) to ManifoldX's `StandardMaterial`, with an `EnvironmentMap` class that supports built-in presets, user-provided HDR/image files, and an optional skybox background.

**Architecture:** `EnvironmentMap` is a pure-Python object (no GPU) built via class methods (`from_hdr`, `from_image`, `from_sky`, `from_color`); `engine.set_environment(env)` triggers CPU precomputation + GPU texture upload. StandardMaterial gains a new `@group(1)` bind group holding the three IBL textures (irradiance cubemap, prefiltered cubemap, BRDF LUT); the `ibl_enabled` and `ibl_intensity` fields in the Globals uniform let the shader skip IBL math when no environment is set, keeping pipeline layout constant.

**Tech Stack:** Python 3.13, NumPy, Pillow (already a `[viz]` dep), wgpu, uv.

# src/manifoldx/ibl.py
"""Image-Based Lighting: EnvironmentMap class, builders, and CPU precomputation."""

from __future__ import annotations

import struct
from typing import Optional, Tuple

import numpy as np


class EnvironmentMap:
    """Equirectangular float32 radiance data (H, W, 3), linear colour space.

    Build via class methods; pass to engine.set_environment().
    Precomputed GPU resources are cached on the object after first upload.
    """

    def __init__(self, data: np.ndarray, intensity: float = 1.0, show_skybox: bool = False):
        assert data.ndim == 3 and data.shape[2] == 3, "data must be (H, W, 3)"
        self.data = data.astype(np.float32)
        self.intensity = intensity
        self.show_skybox = show_skybox

        # Lazily populated by _precompute()
        self._irradiance: Optional[np.ndarray] = None    # (6, 64, 64, 4) float16
        self._prefiltered: Optional[list] = None          # list of 8 arrays (6, S, S, 4) float16
        self._computed: bool = False

    @classmethod
    def from_color(cls, rgb: Tuple[float, float, float]) -> "EnvironmentMap":
        """Uniform ambient — same radiance from all directions."""
        data = np.full((32, 64, 3), rgb, dtype=np.float32)
        return cls(data=data)

    @classmethod
    def from_sky(
        cls,
        zenith: Tuple[float, float, float],
        horizon: Tuple[float, float, float],
        ground: Tuple[float, float, float],
        height: int = 64,
        width: int = 128,
    ) -> "EnvironmentMap":
        """Procedural sky gradient: zenith (top) → horizon (middle) → ground (bottom)."""
        lats = np.linspace(np.pi / 2, -np.pi / 2, height)
        data = np.zeros((height, width, 3), dtype=np.float32)
        z = np.array(zenith, dtype=np.float32)
        h = np.array(horizon, dtype=np.float32)
        g = np.array(ground, dtype=np.float32)
        for i, lat in enumerate(lats):
            t = float(np.sin(lat))  # 1 at zenith, 0 at horizon, -1 at ground
            if t >= 0:
                data[i] = z * t + h * (1.0 - t)
            else:
                data[i] = h * (1.0 + t) + g * (-t)
        return cls(data=data)

    @classmethod
    def from_image(cls, path: str, exposure: float = 1.0) -> "EnvironmentMap":
        """Load any Pillow-supported image (JPEG/PNG) as equirectangular env map.

        Input is treated as sRGB and linearised before use.
        """
        from PIL import Image

        img = Image.open(path).convert("RGB")
        arr = np.array(img, dtype=np.float32) / 255.0
        # sRGB → linear
        arr = np.where(arr <= 0.04045, arr / 12.92, ((arr + 0.055) / 1.055) ** 2.4)
        arr *= exposure
        return cls(data=arr)

    @classmethod
    def from_hdr(cls, path: str) -> "EnvironmentMap":
        """Load a Radiance RGBE (.hdr) file and return linear-float32 data."""
        data = _decode_rgbe(path)
        return cls(data=data)


# Built-in presets — thin wrappers over from_sky / from_color

PRESETS = {
    "studio": lambda: EnvironmentMap.from_sky(
        zenith=(0.6, 0.6, 0.65),
        horizon=(0.5, 0.5, 0.55),
        ground=(0.1, 0.1, 0.1),
    ),
    "sky": lambda: EnvironmentMap.from_sky(
        zenith=(0.05, 0.15, 0.6),
        horizon=(0.5, 0.65, 0.9),
        ground=(0.05, 0.05, 0.05),
    ),
    "neutral": lambda: EnvironmentMap.from_color((0.3, 0.3, 0.3)),
    "dark": lambda: EnvironmentMap.from_color((0.02, 0.02, 0.02)),
}


def _decode_rgbe(path: str) -> np.ndarray:
    """Decode a Radiance RGBE (.hdr) file to float32 (H, W, 3), linear."""
    with open(path, "rb") as f:
        raw = f.read()

    # --- Parse header ---
    pos = 0
    while True:
        nl = raw.index(b"\n", pos)
        line = raw[pos:nl]
        pos = nl + 1
        if line == b"":  # blank line ends header
            break

    # Size line: -Y H +X W
    nl = raw.index(b"\n", pos)
    size_line = raw[pos:nl].decode("ascii")
    pos = nl + 1
    parts = size_line.split()
    H, W = int(parts[1]), int(parts[3])

    # --- Decode scanlines (new RLE format) ---
    result = np.zeros((H, W, 4), dtype=np.uint8)
    for row in range(H):
        marker = raw[pos: pos + 4]
        pos += 4
        if marker[0] == 2 and marker[1] == 2:
            # New RLE: each channel encoded separately
            scanline_w = (marker[2] << 8) | marker[3]
            for ch in range(4):
                x = 0
                while x < scanline_w:
                    code = raw[pos]
                    pos += 1
                    if code > 128:
                        count = code - 128
                        val = raw[pos]
                        pos += 1
                        result[row, x: x + count, ch] = val
                        x += count
                    else:
                        result[row, x: x + code, ch] = list(raw[pos: pos + code])
                        pos += code
                        x += code
        else:
            # Uncompressed / old RLE — treat as raw RGBE
            chunk = np.frombuffer(raw[pos - 4: pos - 4 + W * 4], dtype=np.uint8)
            result[row] = chunk.reshape(W, 4)
            pos += (W - 1) * 4

    # --- RGBE → float32 RGB ---
    R = result[..., 0].astype(np.float32)
    G = result[..., 1].astype(np.float32)
    B = result[..., 2].astype(np.float32)
    E = result[..., 3].astype(np.float32)
    scale = np.where(E > 0, np.ldexp(1.0, E.astype(int) - 128 - 8), 0.0)
    rgb = np.stack([R * scale, G * scale, B * scale], axis=-1)
    return rgb.astype(np.float32)

# ---------------------------------------------------------------------------
# Cubemap face directions (WebGPU convention: +X, -X, +Y, -Y, +Z, -Z)
# ---------------------------------------------------------------------------

def _face_directions(face_size: int):
    """Return a list of 6 (face_size, face_size, 3) world-direction arrays."""
    coords = np.linspace(-1.0, 1.0, face_size, dtype=np.float32)
    u, v = np.meshgrid(coords, coords)
    return [
        np.stack([ np.ones_like(u), -v, -u], axis=-1),  # +X
        np.stack([-np.ones_like(u), -v,  u], axis=-1),  # -X
        np.stack([ u,  np.ones_like(v),  v], axis=-1),  # +Y
        np.stack([ u, -np.ones_like(v), -v], axis=-1),  # -Y
        np.stack([ u, -v,  np.ones_like(u)], axis=-1),  # +Z
        np.stack([-u, -v, -np.ones_like(u)], axis=-1),  # -Z
    ]


def _equirect_to_cubemap(equirect: np.ndarray, face_size: int) -> np.ndarray:
    """Sample an equirectangular image into a 6-face cubemap.

    equirect: (H, W, 3) float32
    Returns: (6, face_size, face_size, 3) float32
    """
    H, W = equirect.shape[:2]
    cube = np.zeros((6, face_size, face_size, 3), dtype=np.float32)
    for fi, dirs in enumerate(_face_directions(face_size)):
        norms = np.linalg.norm(dirs, axis=-1, keepdims=True).clip(min=1e-8)
        d = dirs / norms  # (S, S, 3)
        phi = np.arctan2(d[..., 2], d[..., 0])          # -π..π
        theta = np.arcsin(np.clip(d[..., 1], -1.0, 1.0)) # -π/2..π/2
        eu = (phi / (2 * np.pi) + 0.5) * (W - 1)
        ev = (0.5 - theta / np.pi) * (H - 1)
        # Bilinear sample
        eu0 = np.clip(np.floor(eu).astype(int), 0, W - 1)
        eu1 = np.clip(eu0 + 1, 0, W - 1)
        ev0 = np.clip(np.floor(ev).astype(int), 0, H - 1)
        ev1 = np.clip(ev0 + 1, 0, H - 1)
        wu, wv = (eu - eu0)[..., None], (ev - ev0)[..., None]
        cube[fi] = (
            equirect[ev0, eu0] * (1 - wu) * (1 - wv)
            + equirect[ev0, eu1] * wu * (1 - wv)
            + equirect[ev1, eu0] * (1 - wu) * wv
            + equirect[ev1, eu1] * wu * wv
        )
    return cube


def _sample_cube(cube: np.ndarray, dirs: np.ndarray) -> np.ndarray:
    """Sample a (6, H, W, C) cubemap at N directions (N, 3).

    Returns (N, C) float32.
    """
    # Select face by dominant axis
    ax = np.abs(dirs)
    face_idx = np.argmax(ax, axis=-1)  # 0=X,1=Y,2=Z dominant
    sign = np.sign(dirs[np.arange(len(dirs)), face_idx])
    # Map (face_idx, sign) to cubemap face index
    face_map = np.where(
        face_idx == 0,
        np.where(sign > 0, 0, 1),
        np.where(face_idx == 1, np.where(sign > 0, 2, 3), np.where(sign > 0, 4, 5)),
    )
    # Compute face UV
    S = cube.shape[1]
    u_sel = np.where(
        face_idx == 0,
        np.where(sign > 0, -dirs[:, 2], dirs[:, 2]),
        np.where(face_idx == 1, dirs[:, 0], np.where(sign > 0, dirs[:, 0], -dirs[:, 0])),
    ) / ax[np.arange(len(dirs)), face_idx]
    v_sel = np.where(
        face_idx == 0,
        -dirs[:, 1],
        np.where(face_idx == 1, np.where(sign > 0, dirs[:, 2], -dirs[:, 2]), -dirs[:, 1]),
    ) / ax[np.arange(len(dirs)), face_idx]
    pu = np.clip(((u_sel + 1) / 2) * (S - 1), 0, S - 1).astype(int)
    pv = np.clip(((v_sel + 1) / 2) * (S - 1), 0, S - 1).astype(int)
    return cube[face_map, pv, pu]


def _compute_irradiance(cube: np.ndarray, out_size: int = 64, samples: int = 256) -> np.ndarray:
    """Cosine-weighted hemisphere integration for diffuse irradiance.

    cube: (6, S, S, 3) float32
    Returns: (6, out_size, out_size, 4) float16  (alpha channel = 1)
    """
    irr = np.zeros((6, out_size, out_size, 3), dtype=np.float32)
    rng = np.random.default_rng(0)
    phi_s = rng.uniform(0.0, 2 * np.pi, samples).astype(np.float32)
    cos_t = rng.uniform(0.0, 1.0, samples).astype(np.float32)
    sin_t = np.sqrt(1.0 - cos_t**2)
    # Tangent-space hemisphere samples (cosine-weighted)
    ts = np.stack([sin_t * np.cos(phi_s), sin_t * np.sin(phi_s), cos_t], axis=-1)  # (S, 3)

    for fi, dirs in enumerate(_face_directions(out_size)):
        norms = np.linalg.norm(dirs, axis=-1, keepdims=True).clip(min=1e-8)
        N_all = (dirs / norms).reshape(-1, 3)  # (P, 3), P = out_size^2
        # Build TBN per texel — vectorised
        is_up = np.abs(N_all[:, 1]) > 0.99
        ref = np.where(is_up[:, None], np.array([[1., 0., 0.]]), np.array([[0., 1., 0.]]))
        right = np.cross(N_all, ref)
        right /= np.linalg.norm(right, axis=-1, keepdims=True).clip(min=1e-8)
        up2 = np.cross(N_all, right)
        # World-space sample dirs: (P, samples, 3)
        world = (
            ts[None, :, 0:1] * right[:, None, :]
            + ts[None, :, 1:2] * up2[:, None, :]
            + ts[None, :, 2:3] * N_all[:, None, :]
        ).reshape(-1, 3)  # (P*samples, 3)
        colors = _sample_cube(cube, world).reshape(-1, samples, 3)  # (P, samples, 3)
        irr[fi] = colors.mean(axis=1).reshape(out_size, out_size, 3) * np.pi

    # Add alpha=1 channel and convert to float16 to save GPU memory
    out = np.concatenate([irr, np.ones((*irr.shape[:3], 1), dtype=np.float32)], axis=-1)
    return out.astype(np.float16)


def _ggx_importance_sample(roughness: float, samples: int, rng) -> np.ndarray:
    """Generate GGX-importance-sampled half-vectors in tangent space.

    Returns: (samples, 3) float32 — H vectors in tangent space.
    """
    a = roughness * roughness
    xi1 = rng.random(samples).astype(np.float32)
    xi2 = rng.random(samples).astype(np.float32)
    phi = 2 * np.pi * xi1
    cos_theta = np.sqrt((1 - xi2) / np.maximum(1 - (1 - a**2) * xi2, 1e-8))
    sin_theta = np.sqrt(np.maximum(1 - cos_theta**2, 0))
    H = np.stack([sin_theta * np.cos(phi), sin_theta * np.sin(phi), cos_theta], axis=-1)
    return H.astype(np.float32)


def _compute_prefiltered(cube: np.ndarray, roughness: float, out_size: int, samples: int = 512) -> np.ndarray:
    """GGX-importance-sampled prefiltered environment map at given roughness.

    cube: (6, S, S, 3) float32
    Returns: (6, out_size, out_size, 4) float16
    """
    pf = np.zeros((6, out_size, out_size, 3), dtype=np.float32)
    rng = np.random.default_rng(42)
    H_ts = _ggx_importance_sample(max(roughness, 0.01), samples, rng)  # (samples, 3)

    for fi, dirs in enumerate(_face_directions(out_size)):
        norms = np.linalg.norm(dirs, axis=-1, keepdims=True).clip(min=1e-8)
        N_all = (dirs / norms).reshape(-1, 3)  # (P, 3)
        is_up = np.abs(N_all[:, 1]) > 0.99
        ref = np.where(is_up[:, None], np.array([[1., 0., 0.]]), np.array([[0., 1., 0.]]))
        right = np.cross(N_all, ref)
        right /= np.linalg.norm(right, axis=-1, keepdims=True).clip(min=1e-8)
        up2 = np.cross(N_all, right)
        # Half-vectors in world space: (P, samples, 3)
        H_world = (
            H_ts[None, :, 0:1] * right[:, None, :]
            + H_ts[None, :, 1:2] * up2[:, None, :]
            + H_ts[None, :, 2:3] * N_all[:, None, :]
        )  # (P, samples, 3)
        # Reflection direction: assume V ≈ N (split-sum approximation)
        N_exp = N_all[:, None, :]  # (P, 1, 3)
        NdotH = np.clip((H_world * N_exp).sum(axis=-1, keepdims=True), 0, 1)
        L = 2 * NdotH * H_world - N_exp  # (P, samples, 3)
        L_flat = L.reshape(-1, 3)
        colors = _sample_cube(cube, L_flat).reshape(-1, samples, 3)  # (P, samples, 3)
        NdotL = np.clip((L * N_exp).sum(axis=-1, keepdims=True), 0, 1)  # (P, samples, 1)
        weighted = (colors * NdotL).sum(axis=1) / NdotL.sum(axis=1).clip(min=1e-4)
        pf[fi] = weighted.reshape(out_size, out_size, 3)

    out = np.concatenate([pf, np.ones((*pf.shape[:3], 1), dtype=np.float32)], axis=-1)
    return out.astype(np.float16)

```python
#!/usr/bin/env python3
"""Offline Smith-GGX BRDF LUT generator (split-sum, 512x512, 1024 samples per texel).

Usage: uv run python scripts/gen_brdf_lut.py
Output: src/manifoldx/assets/ibl/brdf_lut.npy  — (512, 512, 2) float32
"""
import numpy as np
from pathlib import Path

OUT = Path(__file__).parent.parent / "src/manifoldx/assets/ibl/brdf_lut.npy"
SIZE = 512
SAMPLES = 1024

rng = np.random.default_rng(0)


def geometry_schlick_ggx(ndotv: np.ndarray, roughness: float) -> np.ndarray:
    k = (roughness * roughness) / 2.0
    return ndotv / (ndotv * (1 - k) + k)


def geometry_smith(ndotv: np.ndarray, ndotl: np.ndarray, roughness: float) -> np.ndarray:
    return geometry_schlick_ggx(ndotv, roughness) * geometry_schlick_ggx(ndotl, roughness)


lut = np.zeros((SIZE, SIZE, 2), dtype=np.float32)

for j in range(SIZE):
    roughness = (j + 0.5) / SIZE
    a = roughness * roughness
    xi1 = rng.random(SAMPLES).astype(np.float32)
    xi2 = rng.random(SAMPLES).astype(np.float32)
    # GGX importance sample H
    cos_theta = np.sqrt((1 - xi2) / np.maximum(1 - (1 - a**2) * xi2, 1e-8))
    sin_theta = np.sqrt(np.maximum(1 - cos_theta**2, 0))
    phi = 2 * np.pi * xi1
    H = np.stack([sin_theta * np.cos(phi), sin_theta * np.sin(phi), cos_theta], axis=-1)

    for i in range(SIZE):
        NdotV = (i + 0.5) / SIZE
        V = np.array([np.sqrt(1 - NdotV**2), 0.0, NdotV], dtype=np.float32)  # (3,)
        # Reflect over H for all samples
        VdotH = np.clip((V[None, :] * H).sum(axis=-1), 0, 1)  # (SAMPLES,)
        L = 2 * VdotH[:, None] * H - V[None, :]  # (SAMPLES, 3)
        NdotL = np.clip(L[:, 2], 0, 1)
        NdotH = np.clip(H[:, 2], 0, 1)
        mask = NdotL > 0
        if mask.sum() == 0:
            continue
        G = geometry_smith(NdotV, NdotL[mask], roughness)
        G_vis = G * VdotH[mask] / np.maximum(NdotH[mask] * NdotV, 1e-8)
        Fc = (1 - VdotH[mask]) ** 5
        lut[i, j, 0] += (G_vis * (1 - Fc)).sum() / SAMPLES
        lut[i, j, 1] += (G_vis * Fc).sum() / SAMPLES

lut = np.clip(lut, 0, 1)
OUT.parent.mkdir(parents=True, exist_ok=True)
np.save(OUT, lut)
print(f"Saved BRDF LUT to {OUT}  shape={lut.shape}")
```

        device.queue.write_texture(
            {"texture": tex, "mip_level": 0, "origin": (0, 0, 0)},
            data,
            {"bytes_per_row": 8, "rows_per_image": 1},
            (1, 1, 6),
        )
        view = tex.create_view(
            format=wgpu.TextureFormat.rgba16float,
            dimension=wgpu.TextureViewDimension.cube,
            base_mip_level=0,
            mip_level_count=1,
            base_array_layer=0,
            array_layer_count=6,
        )
        return view

    def _create_neutral_lut(self, device):
        """512×512 rg16float LUT texture + view filled with 0.5."""
        from manifoldx.ibl import load_brdf_lut
        lut_data = load_brdf_lut()  # (512, 512, 2) float32
        # Convert to float16 for upload
        lut16 = lut_data.astype(np.float16)
        tex = device.create_texture(
            size=(512, 512, 1),
            format=wgpu.TextureFormat.rg16float,
            usage=wgpu.TextureUsage.TEXTURE_BINDING | wgpu.TextureUsage.COPY_DST,
            dimension=wgpu.TextureDimension.d2,
            mip_level_count=1,
            sample_count=1,
        )
        device.queue.write_texture(
            {"texture": tex, "mip_level": 0, "origin": (0, 0, 0)},
            lut16.tobytes(),
            {"bytes_per_row": 512 * 4, "rows_per_image": 512},
            (512, 512, 1),
        )
        view = tex.create_view(
            format=wgpu.TextureFormat.rg16float,
            dimension=wgpu.TextureViewDimension.d2,
        )
        return view

    def _create_ibl_bind_group(self, device, irr_view, pf_view, lut_view):
        return device.create_bind_group(
            layout=self._ibl_placeholder_bind_group_layout,
            entries=[
                {"binding": 0, "resource": self._ibl_sampler},
                {"binding": 1, "resource": irr_view},
                {"binding": 2, "resource": pf_view},
                {"binding": 3, "resource": lut_view},
            ],
        )

    def _upload_ibl_env(self, device, env):
        """Upload EnvironmentMap precomputed data to GPU textures."""
        env._precompute()
        irr = env._irradiance   # (6, 64, 64, 4) float16
        pf0 = env._prefiltered  # list of 8 arrays

        # Irradiance cubemap: 64×64×6
        irr_tex = device.create_texture(
            size=(64, 64, 6),
            format=wgpu.TextureFormat.rgba16float,
            usage=wgpu.TextureUsage.TEXTURE_BINDING | wgpu.TextureUsage.COPY_DST,
            dimension=wgpu.TextureDimension.d2,
            mip_level_count=1,
            sample_count=1,
        )
        device.queue.write_texture(
            {"texture": irr_tex, "mip_level": 0, "origin": (0, 0, 0)},
            irr.tobytes(),
            {"bytes_per_row": 64 * 8, "rows_per_image": 64},
            (64, 64, 6),
        )
        irr_view = irr_tex.create_view(
            format=wgpu.TextureFormat.rgba16float,
            dimension=wgpu.TextureViewDimension.cube,
            base_mip_level=0, mip_level_count=1,
            base_array_layer=0, array_layer_count=6,
        )

        # Prefiltered cubemap: 128×128×6 with 8 mips
        pf_tex = device.create_texture(
            size=(128, 128, 6),
            format=wgpu.TextureFormat.rgba16float,
            usage=wgpu.TextureUsage.TEXTURE_BINDING | wgpu.TextureUsage.COPY_DST,
            dimension=wgpu.TextureDimension.d2,
            mip_level_count=8,
            sample_count=1,
        )
        for mip, pf in enumerate(pf0):
            s = max(1, 128 >> mip)  # mip size
            device.queue.write_texture(
                {"texture": pf_tex, "mip_level": mip, "origin": (0, 0, 0)},
                pf.astype(np.float16).tobytes(),
                {"bytes_per_row": s * 8, "rows_per_image": s},
                (s, s, 6),
            )
        pf_view = pf_tex.create_view(
            format=wgpu.TextureFormat.rgba16float,
            dimension=wgpu.TextureViewDimension.cube,
            base_mip_level=0, mip_level_count=8,
            base_array_layer=0, array_layer_count=6,
        )

        lut_view = self._ibl_active_lut_tex or self._create_neutral_lut(device)
        # Reuse the static BRDF LUT view from placeholder setup
        lut_view = self._ibl_neutral_lut_view

        self._ibl_active_bind_group = self._create_ibl_bind_group(
            device, irr_view, pf_view, lut_view
        )
        self._ibl_env_id = id(env)
```

"""Skybox render pass — draws the prefiltered environment map as scene background."""

import wgpu

_SKYBOX_SHADER = """
struct Globals {
    vp:            mat4x4<f32>,
    view:          mat4x4<f32>,
    proj:          mat4x4<f32>,
    camera_pos:    vec3<f32>,
    _pad0:         f32,
    viewport_size: vec2<f32>,
    _pad1:         vec2<f32>,
    ibl_intensity: f32,
    ibl_enabled:   u32,
    _pad_ibl:      vec2<f32>,
};

@group(0) @binding(0) var<uniform> globals: Globals;
@group(1) @binding(0) var env_sampler:    sampler;
@group(1) @binding(2) var prefiltered_map: texture_cube<f32>;

struct VertexOutput {
    @builtin(position) pos: vec4<f32>,
    @location(0) dir: vec3<f32>,
};

@vertex
fn vs_main(@builtin(vertex_index) vi: u32) -> VertexOutput {
    // Fullscreen triangle
    var positions = array<vec2<f32>, 3>(
        vec2<f32>(-1.0, -1.0),
        vec2<f32>( 3.0, -1.0),
        vec2<f32>(-1.0,  3.0),
    );
    let p = positions[vi];
    // Reconstruct world direction from NDC
    let clip = vec4<f32>(p, 0.9999, 1.0);
    // Inverse VP to get world ray direction
    let inv_proj = ... // passed via globals or pre-baked
    var out: VertexOutput;
    out.pos = vec4<f32>(p, 1.0, 1.0);  // z=w → depth = 1.0 (far plane)
    out.dir = (globals.view * vec4<f32>(p.x / globals.proj[0][0],
                                        p.y / globals.proj[1][1],
                                        -1.0, 0.0)).xyz;
    return out;
}

@fragment
fn fs_main(in: VertexOutput) -> @location(0) vec4<f32> {
    let color = textureSampleLevel(prefiltered_map, env_sampler, normalize(in.dir), 0.0).rgb;
    return vec4<f32>(pow(color / (color + 1.0), vec3<f32>(1.0 / 2.2)), 1.0);
}
"""


def render_skybox(rp, engine, render_pass):
    """Render environment map as background (before mesh pass).

    Only called when engine.environment is not None and show_skybox is True.
    The skybox renders at depth=1.0 (far plane) so all scene geometry draws
    in front without any extra sorting.
    """
    env = engine.environment
    if env is None or not env.show_skybox:
        return

    if id(env) != rp._ibl_env_id:
        rp._upload_ibl_env(rp._device, env)

    key = "skybox"
    if key not in rp._pipelines:
        # Globals bind group layout (group 0) — reuse if already created,
        # else create a minimal one: just the globals uniform.
        bg0_layout = rp._device.create_bind_group_layout(entries=[
            {
                "binding": 0,
                "visibility": wgpu.ShaderStage.VERTEX | wgpu.ShaderStage.FRAGMENT,
                "buffer": {"type": wgpu.BufferBindingType.uniform},
            }
        ])
        pipeline_layout = rp._device.create_pipeline_layout(
            bind_group_layouts=[bg0_layout, rp._ibl_placeholder_bind_group_layout]
        )
        shader = rp._device.create_shader_module(code=_SKYBOX_SHADER)
        pipeline = rp._device.create_render_pipeline(
            layout=pipeline_layout,
            vertex={"module": shader, "entry_point": "vs_main", "buffers": []},
            primitive={"topology": wgpu.PrimitiveTopology.triangle_list},
            depth_stencil={
                "format": wgpu.TextureFormat.depth24plus,
                "depth_write_enabled": False,
                "depth_compare": wgpu.CompareFunction.less_equal,
            },
            fragment={
                "module": shader,
                "entry_point": "fs_main",
                "targets": [{"format": engine._texture_format}],
            },
        )
        rp._pipelines[key] = pipeline
        rp._skybox_bg0_layout = bg0_layout

    pipeline = rp._pipelines[key]
    bg0 = rp._device.create_bind_group(
        layout=rp._skybox_bg0_layout,
        entries=[{
            "binding": 0,
            "resource": {"buffer": rp._globals_buffer, "offset": 0, "size": 240},
        }],
    )
    ibl_bg = rp._ibl_active_bind_group or rp._ibl_placeholder_bind_group

    render_pass.set_pipeline(pipeline)
    render_pass.set_bind_group(0, bg0)
    render_pass.set_bind_group(1, ibl_bg)
    render_pass.draw(3, 1, 0, 0)
```

**Important note on the WGSL skybox vertex shader:** The view direction reconstruction uses `globals.proj[0][0]` (focal length X) and `globals.proj[1][1]` (focal length Y) to unproject from NDC. The `globals.view` matrix is used to rotate the ray into world space. This avoids needing an inverse matrix on the GPU; check that `globals.view` and `globals.proj` are column-major in WGSL (they are — the renderer transposes before upload).

    for col in range(4):
